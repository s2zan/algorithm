# 11437 정점들의 거리

[문제 링크](https://www.acmicpc.net/problem/11437)

### 내 풀이

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        int n = Integer.parseInt(br.readLine());
        int[] parent = new int[n];
        int[] depth = new int[n];
        int[] distance = new int[n];
        List<HashMap<Integer, Integer>> connect = new ArrayList<>();
        for(int i = 0; i < n; i++){
            connect.add(new HashMap<>());
            parent[i] = -1;
            depth[i] = -1;
            distance[i] = 0;
        }

        String[] line;
        for(int i = 0; i < n - 1; i++){
            line = br.readLine().split(" ");
            connect.get(Integer.parseInt(line[0])-1).put(Integer.parseInt(line[1])-1, Integer.parseInt(line[2]));
            connect.get(Integer.parseInt(line[1])-1).put(Integer.parseInt(line[0])-1, Integer.parseInt(line[2]));
        }

        Queue<Integer> queue = new LinkedList<>();
        queue.offer(0);
        depth[0] = 0;
        while(!queue.isEmpty()){
            int x = queue.poll();
            for(Integer y: connect.get(x).keySet()){
                if(depth[y] < 0){
                    depth[y] = depth[x] + 1;
                    parent[y] = x;
                    distance[y] = connect.get(x).get(y);
                    queue.offer(y);
                }
            }
        }
      
        int m = Integer.parseInt(br.readLine());
        for(int i = 0; i < m; i++){
            line = br.readLine().split(" ");
            int x = Integer.parseInt(line[0]) - 1;
            int y = Integer.parseInt(line[1]) - 1;
            int dist = 0;

            if(depth[x] > depth[y]){
                int temp = x;
                x = y;
                y = temp;
            }

            while(depth[y] != depth[x]){
                dist += distance[y];
                y = parent[y];
            }

            while(x != y){
                dist += distance[x] + distance[y];
                x = parent[x];
                y = parent[y];
            }
            System.out.println(dist);
        }
    }
}
```
