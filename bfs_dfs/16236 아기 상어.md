# 16236 아기 상어

[문제 링크](https://www.acmicpc.net/problem/16236)

### 내 코드
```python
from queue import Queue


class Shark:
    def __init__(self):
        self.x = 0
        self.y = 0
        self.size = 2


dx = [-1, 0, 0, 1]
dy = [0, -1, 1, 0]


def feed(m, shark, n):
    q = Queue()
    q.put([shark.x, shark.y, 0])
    is_visit = [[False] * n for _ in range(n)]
    is_visit[shark.x][shark.y] = True
    candidate = []
    flag = -1
    while q.qsize() > 0:
        x, y, dist = q.get()
        for xx, yy in zip(dx, dy):
            if 0<=xx+x<n and 0<=yy+y<n and not is_visit[xx+x][yy+y]:
                if flag != -1 and flag < dist + 1:
                    break
                if 0 < m[xx+x][yy+y] < shark.size:
                    if flag == -1:
                        flag = dist + 1
                    candidate.append([xx+x,yy+y])
                if m[xx+x][yy+y] <= shark.size:
                    q.put([xx+x, yy+y, dist + 1])
                is_visit[xx+x][yy+y] = True
    if len(candidate) > 0:
        candidate.sort(key=lambda a: (a[0], a[1]))
        return candidate[0][0], candidate[0][1], flag
    return -1, -1, -1


n = int(input())
m = [[] for _ in range(n)]
shark = Shark()
for i in range(n):
    temp = list(map(int, input().split()))
    for j, t in enumerate(temp):
        if t == 9:
            temp[j] = 0
            shark.x = i
            shark.y = j
    m[i] = temp
time = 0
eat = 0
while True:
    x, y, dist = feed(m, shark, n)
    m[x][y] = 0
    if x == -1:
        break
    time += dist
    shark.x = x
    shark.y = y
    eat += 1
    if eat == shark.size:
        shark.size += 1
        eat = 0
print(time)
```
