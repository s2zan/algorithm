##### Python

```python
for t in range(10):
    l = int(input())
    origin = input().split()
    n = int(input())
    command = input().split()

    idx = 0
    while idx < len(command):
        type = command[idx]
        if type == 'I':
            x = int(command[idx + 1])
            y = int(command[idx + 2])
            s = command[idx + 3:idx + y + 3]
            origin = origin[:x] + s + origin[x:]
            idx += y + 3
        elif type == 'D':
            x = int(command[idx + 1])
            y = int(command[idx + 2])
            origin = origin[:x + 1] + origin[x + 1 + y:]
            idx += 3
        elif type == 'A':
            y = int(command[idx + 1])
            s = command[idx + 2:idx + y + 2]
            origin = origin + s
            idx += y + 2
    print("#%d"%(t+1), end=' ')
    print(" ".join(origin[:10]))
```

##### Java

```java
import java.io.*;
import java.util.Arrays;
import java.util.LinkedList;

public class Solution {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        for(int t = 1; t <= 10; t++){
            int l = Integer.parseInt(br.readLine());
            LinkedList<String> crypto= new LinkedList<>(Arrays.asList(br.readLine().split(" ")));
            int n = Integer.parseInt(br.readLine());
            String[] command = br.readLine().split(" ");

            int idx = 0;
            int x, y;
            while(idx < command.length){
                String type = command[idx++];
                switch (type){
                    case "I":
                        x = Integer.parseInt(command[idx++]);
                        y = Integer.parseInt(command[idx++]);

                        for(int i = 0; i < y; i++) {
                            crypto.add(x + i, command[idx + i]);
                        }
                        idx += y;
                        break;
                    case "D":
                        x = Integer.parseInt(command[idx++]);
                        y = Integer.parseInt(command[idx++]);
                        for(int i = 0; i < y; i++) {
                            crypto.remove(x + 1);
                        }
                        break;
                    case "A":
                        y = Integer.parseInt(command[idx++]);
                        for(int i = 0; i < y; i++) {
                            crypto.add(command[idx + i]);
                        }
                        idx += y;
                        break;
                }
            }
            System.out.printf("#%d ", t);
            for(int i = 0; i < 10; i++) System.out.printf("%s ", crypto.get(i));
            System.out.println();
        }
    }
}
```
